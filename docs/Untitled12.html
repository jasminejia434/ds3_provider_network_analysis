import networkx as nx
import matplotlib.pyplot as plt
import pandas as pd
from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler

# Take a sample of the data to reduce computation time
sample_size = 3000
df = df.sample(n=sample_size, random_state=42)

# Extract NPI and TIN_VALUE columns
npi_list = df['NPI'].astype(str).tolist()
tin_list = df['TIN_VALUE'].astype(str).tolist()

# Create a bipartite graph
B = nx.Graph()

# Add nodes with the bipartite attribute
B.add_nodes_from(npi_list, bipartite='NPI')
B.add_nodes_from(tin_list, bipartite='TIN')

# Add edges between NPIs and TINs
edges = list(zip(npi_list, tin_list))
B.add_edges_from(edges)

print(B)

# Separate nodes by set
npi_nodes = {n for n, d in B.nodes(data=True) if d['bipartite'] == 'NPI'}
tin_nodes = set(B) - npi_nodes

# Clustering analysis for NPIs and TINs
# Create a dataframe for clustering with NPI and TIN relationships
adjacency_matrix = nx.to_pandas_adjacency(B, nodelist=list(npi_nodes) + list(tin_nodes))

# Limit the size of the adjacency matrix to avoid issues with large graph size
adjacency_matrix = adjacency_matrix.loc[:, (adjacency_matrix != 0).any(axis=0)]

# Standardize the data
scaler = StandardScaler()
adjacency_matrix_scaled = scaler.fit_transform(adjacency_matrix)

# Elbow plot to determine optimal number of clusters
sse = []
k_range = range(1, 11)
for k in k_range:
    kmeans = KMeans(n_clusters=k, random_state=42)
    kmeans.fit(adjacency_matrix_scaled)
    sse.append(kmeans.inertia_)

plt.figure(figsize=(10, 6))
plt.plot(k_range, sse, marker='o')
plt.xlabel('Number of Clusters')
plt.ylabel('Sum of Squared Errors (SSE)')
plt.title('Elbow Plot to Determine Optimal Number of Clusters')
plt.tight_layout()
plt.show()
